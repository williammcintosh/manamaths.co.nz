{\LARGE \textbf{Area of L-shapes and composite shapes — Proficient}}
\begin{tasks}(3)
\task {A composite shape consists of a 6 m × 4 m rectangle attached to a right triangle with legs 3 m and 4 m. Find total area.}
\task {Find the area of an L‑shaped garden where the outer dimensions are 12 m by 9 m and the cut‑out is a 5 m by 4 m rectangle.}
\task {A shape is made by joining a 7 m × 5 m rectangle and a semicircle of diameter 5 m. Find the total area (use \mbox{$\pi\approx 3.14$}).}
\task {A room has an L‑shape with missing corner; the outer rectangle is 14 m by 10 m, the missing corner is a triangle of base 6 m and height 4 m. Find area.}
\task {Draw a line to split the shape into a rectangle and a triangle, then calculate area.}
\task {A composite shape consists of a 8 m × 6 m rectangle with a 3 m × 2 m rectangle removed from one corner, and a 2 m × 2 m square added on the opposite side. Find area.}
\task {A garden is shaped like a rectangle joined to a trapezium. Rectangle: 10 m by 4 m. Trapezium: parallel sides 4 m and 6 m, height 3 m. Find total area.}
\task {A pathway forms an L around a building; the outer dimensions are 20 m by 15 m, and the inner cut‑out is a rectangle 12 m by 8 m. Find area of the pathway.}
\task {Find the area of a shape composed of a 9 m × 5 m rectangle and a quarter‑circle of radius 3 m attached to one corner.}
\task {A swimming pool has a composite shape: a 12 m × 8 m rectangle with a 4 m × 4 m square cut out, and a semicircular end of radius 2 m. Find area.}
\task {Draw a line to split the shape into a rectangle and a right triangle, then calculate area given side lengths.}
\task {A flower bed consists of a 5 m × 3 m rectangle attached to a triangle with base 3 m and height 2 m. Find total area.}
\end{tasks}
