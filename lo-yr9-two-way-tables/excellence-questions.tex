\begin{tasks}(3)
\task How many students are there altogether?
\[
\begin{array}{|l|c|c|c|}
\hline
 & \text{Boys} & \text{Girls} & \text{Total} \\\hline
\text{Apples} & 7 & 8 & 15 \\\hline
\text{Bananas} & 5 & 10 & 15 \\\hline
\text{Total} & 12 & 18 & 30 \\\hline
\end{array}
\]

\task Use the same apples/bananas table. Write the probability a random student likes apples.

\task Use the same apples/bananas table. Write the probability a random student is a girl who likes bananas.

\task The football row total is \mbox{$15$}. How many students are not in the football row?
\[
\begin{array}{|l|c|c|c|}
\hline
 & \text{Boys} & \text{Girls} & \text{Total} \\\hline
\text{Football} & 8 & 7 & 15 \\\hline
\text{Netball} & 9 & 12 & 21 \\\hline
\text{Total} & 17 & 19 & 36 \\\hline
\end{array}
\]

\task One column total is \mbox{$20$}. What is the other column total?
\[
\begin{array}{|l|c|c|c|}
\hline
 & \text{Boys} & \text{Girls} & \text{Total} \\\hline
\text{Football} & 8 & 7 & 15 \\\hline
\text{Netball} & 12 & 9 & 21 \\\hline
\text{Total} & 20 & 16 & 36 \\\hline
\end{array}
\]

\task Write the probability a random student has a pet.
\[
\begin{array}{|l|c|c|c|}
\hline
 & \text{Boys} & \text{Girls} & \text{Total} \\\hline
\text{Has pets} & 11 & 9 & 20 \\\hline
\text{No pets} & 5 & 7 & 12 \\\hline
\text{Total} & 16 & 16 & 32 \\\hline
\end{array}
\]

\task Using the same pets table, write the probability a random student is a girl without a pet.

\task Using the same pets table, which is more likely: having a pet or not having a pet?

\task A student says the total of all row totals should match the total of all column totals. Are they correct? Explain.

\begin{array}{|l|c|c|c|}

\text{Has pets} & 11 & 9 & 20 \\\hline
\text{No pets} & 5 & 7 & 12 \\\hline
\text{Total} & 16 & 16 & 32 \\\hline
\end{array}

\task A student says the denominator in a table probability should always be a row total. Are they correct? Explain.
\[
\begin{array}{|l|c|c|c|}
\hline
