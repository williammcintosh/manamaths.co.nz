\begin{tasks}(3)
\task A student says \mbox{$x\times x^2=x^2$}. Are they correct? Explain.
\task A student says \mbox{$3a\times 4a=12a$}. Are they correct? Explain.
\task A student says \mbox{$2m^2\times 5m=10m^3$}. Are they correct? Explain.
\task Fill in the blank: \mbox{$\square p\times 3p^2=24p^3$}.
\task Fill in the blank: \mbox{$5q^2\times \square q=35q^3$}.
\task Fill in the blank: \mbox{$4r\times 2r\times r=\square r^3$}.
\task Complete: \mbox{$6x^2\times 2x=\rule{2cm}{0.15mm}$}.
\task Complete: \mbox{$3y\times y\times 4y=\rule{2cm}{0.15mm}$}.
\task Which is greater: \mbox{$2a\times 7a^2$} or \mbox{$3a^2\times 4a$}? Show enough working.
\task Which is greater: \mbox{$5m\times 5m$} or \mbox{$2m^2\times 12$}? Show enough working.
\task Write two different multiplications that both simplify to \mbox{$12x^3$}.
\task Write two different multiplications that both simplify to \mbox{$18a^2$}.
\task Find a missing factor: \mbox{$\square\times 4b^2=28b^3$}.
\task Find a missing factor: \mbox{$9c\times \square=27c^2$}.
\task Explain why \mbox{$2x\times 3x\times x$} becomes \mbox{$6x^2$} or \mbox{$6x^3$}.
\task Compare \mbox{$4p\times 3p^2$} with \mbox{$5p^2\times 2p$}. Which product is greater?
\task Is \mbox{$7k\times k^2=7k^3$} correct? Give a short reason.
\task Is \mbox{$4n^2\times 2n^2=8n^4$} correct? Give a short reason.
\end{tasks}
