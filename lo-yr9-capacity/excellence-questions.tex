{\LARGE \textbf{Capacity — Excellence}}
\begin{tasks}(3)
\task A cafe makes \mbox{$18$} smoothies. Each smoothie uses \mbox{$375\text{ mL}$} of milk. What volume of milk is needed in litres?
\task A fish tank holds \mbox{$24\text{ L}$} when full. It is \mbox{$62.5\%$} full. How many millilitres of water are in the tank?
\task A drum contains \mbox{$3.6\text{ L}$} of cleaner. It is poured equally into \mbox{$8$} spray bottles. What volume goes into each bottle in millilitres?
\task A recipe needs \mbox{$2.4\text{ L}$} of soup. You have one \mbox{$1\text{ L}$} carton and three \mbox{$450\text{ mL}$} cartons. Do you have enough? Show the total in litres.
\task A water cooler starts with \mbox{$15\text{ L}$}. During the day, \mbox{$3.25\text{ L}$} is used and then \mbox{$1800\text{ mL}$} is added. How much water is in the cooler now?
\task A laboratory mixes \mbox{$0.85\text{ L}$} of solution A with \mbox{$650\text{ mL}$} of solution B and \mbox{$1.2\text{ L}$} of water. Find the total volume in litres and millilitres.
\task Which is greater, \mbox{$0.6\text{ L}$} or \mbox{$580\text{ mL}$}? Show a conversion to justify your answer.
\task A container is labelled \mbox{$2500\text{ cm}^3$}. Using \mbox{$1\text{ cm}^3=1\text{ mL}$}, state its capacity in millilitres and litres.
\end{tasks}
