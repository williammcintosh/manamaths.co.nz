\begin{tasks}(3)

\task Convert \mbox{$2750\text{ mL}$} to litres.

\task A measuring jug contains \mbox{$1850\text{ mL}$}. Write this in litres.

\task Are \mbox{$1.25\text{ L}$} and \mbox{$1250\text{ mL}$} equal? Explain.
