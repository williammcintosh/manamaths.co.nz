{\LARGE \textbf{Foundation}}
\begin{tasks}(3)
\task A coin is tossed \mbox{$20$} times and lands heads \mbox{$11$} times. Estimate the probability of heads.
\task A die is rolled \mbox{$30$} times and a \mbox{$6$} appears \mbox{$4$} times. Estimate the probability of rolling a \mbox{$6$}.
\task A spinner lands on red \mbox{$9$} times in \mbox{$15$} spins. Estimate the probability of red.
\task In \mbox{$25$} draws from a bag, a blue counter is picked \mbox{$10$} times. Estimate the probability of blue.
\task A survey chart shows \mbox{$18$} of \mbox{$30$} students walk to school. Estimate the probability that a randomly chosen student walks to school.
\task A weather record shows rain on \mbox{$6$} of the last \mbox{$10$} days. Estimate the probability of rain tomorrow using this data.
\task A coin lands tails \mbox{$14$} times in \mbox{$20$} tosses. Write the experimental probability of tails.
\task A spinner lands on green \mbox{$7$} times in \mbox{$28$} spins. Write the experimental probability of green.
\task Out of \mbox{$40$} penalty shots, \mbox{$26$} are scored. Estimate the probability of scoring.
\task Out of \mbox{$50$} seeds planted, \mbox{$35$} sprout. Estimate the probability that a seed sprouts.
\task Which is greater: an experimental probability of \mbox{$\frac{9}{20}$} or \mbox{$\frac{1}{2}$}?
\task Which is smaller: \mbox{$30\%$} success or \mbox{$\frac{2}{5}$} success?
\task Fill in the blank: \mbox{$12$} wins from \mbox{$20$} games gives an experimental probability of \mbox{$\frac{\square}{20}$}.
\task Fill in the blank: \mbox{$15$} successes from \mbox{$25$} trials gives an experimental probability of \mbox{$\square\%$}.
\task A student says \mbox{$3$} reds in \mbox{$12$} draws means the probability of red is \mbox{$\frac{1}{3}$}. Are they correct?
\task Explain in one short sentence what experimental probability means.
\end{tasks}
