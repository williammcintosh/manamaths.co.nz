\begin{tasks}(3)
\task A coin is tossed \mbox{$80$} times and lands heads \mbox{$37$} times. Estimate the probability of heads as a fraction and decimal.
\task A die is rolled \mbox{$60$} times and an even number appears \mbox{$34$} times. Estimate the probability of rolling an even number.
\task A spinner lands on blue \mbox{$18$} times in \mbox{$45$} spins. Estimate the probability of blue.
\task A bus is late on \mbox{$9$} of the last \mbox{$24$} school days. Estimate the probability it is late on a future school day.
\task In a tally chart, \mbox{$14$} of \mbox{$35$} customers choose mango. Estimate the probability that the next customer chooses mango.
\task A class bar chart shows \mbox{$12$} students play football, \mbox{$9$} netball, \mbox{$6$} basketball, and \mbox{$3$} hockey. Estimate the probability a randomly chosen student plays netball.
\task A bag is sampled \mbox{$50$} times with replacement and a yellow counter appears \mbox{$21$} times. Estimate the probability of yellow.
\task In \mbox{$120$} free throws, a player scores \mbox{$78$}. Estimate the probability of scoring a free throw.
\task Which is closer to \mbox{$\frac{1}{2}$}: \mbox{$24$} successes in \mbox{$50$} trials or \mbox{$17$} successes in \mbox{$30$} trials?
\task Fill in the blank: \mbox{$48$} correct answers from \mbox{$60$} questions gives an experimental probability of \mbox{$\square\%$}.
\end{tasks}
