{\LARGE \textbf{Excellence}}
\begin{tasks}(3)
\task A student says \mbox{$P(\text{not even on a fair die})=\frac{2}{6}$}. Are they correct? Explain.
\task A student says \mbox{$P(\text{at least one head in two coin tosses})=\frac{2}{4}$}. Are they correct? Explain.
\task A fair die is rolled. Write \mbox{$P(\text{prime})$} in F/T form and simplify it.
\task A fair die is rolled. Write \mbox{$P(\text{factor of }6)$} in F/T form and simplify it.
\task A bag has \mbox{$4$} red, \mbox{$5$} blue, and \mbox{$3$} green counters. Write the probability of red or green in F/T form.
\task A bag has \mbox{$2$} gold, \mbox{$7$} silver, and \mbox{$1$} bronze token. Write the probability of not picking silver in F/T form.
\task A standard deck card is chosen. Write the probability of drawing a red king in F/T form.
\task A standard deck card is chosen. Write the probability of drawing a non-face card in F/T form.
\task A fair spinner has equal sectors labelled \mbox{$1$} to \mbox{$8$}. Write the probability of landing on a factor of \mbox{$8$} in F/T form.
\task A fair spinner has equal sectors labelled \mbox{$1$} to \mbox{$10$}. Write the probability of landing on a multiple of \mbox{$3$} in F/T form.
\task A coin is tossed and a fair die is rolled. Write the probability of getting heads and a \mbox{$5$} in F/T form.
\task A coin is tossed and a fair die is rolled. Write the probability of getting tails and an even number in F/T form.
\task Fill in the blank: if \mbox{$P(\text{event})=\frac{5}{12}$}, then the number of favourable outcomes could be \rule{1.4cm}{0.15mm} when the total outcomes are \mbox{$12$}.
\task Fill in the blank: if a bag has \mbox{$9$} counters and \mbox{$P(\text{red})=\frac{2}{9}$}, then there are \rule{1.4cm}{0.15mm} red counters.
\task Which is greater: \mbox{$P(\text{rolling a number less than }5)$} on a fair die or \mbox{$P(\text{drawing a club})$} from a standard deck? Show enough working to justify.
\task Which is smaller: \mbox{$P(\text{factor of }12)$} on a fair die or \mbox{$P(\text{exactly one head})$} in two coin tosses? Explain.
\task Which does not belong: \mbox{$\frac{1}{2}$}, \mbox{$\frac{3}{6}$}, \mbox{$\frac{2}{4}$}, \mbox{$\frac{2}{3}$}?
\task Explain why an F/T probability can never be greater than \mbox{$1$}.
\end{tasks}
