{\LARGE \textbf{Foundation}}
\begin{tasks}(3)
\task Write the probability of getting heads on a fair coin in F/T form.
\task Write the probability of getting tails on a fair coin in F/T form.
\task Write the probability of rolling a \mbox{$6$} on a fair die in F/T form.
\task Write the probability of rolling an even number on a fair die in F/T form.
\task Write the probability of rolling a number less than \mbox{$3$} on a fair die in F/T form.
\task Write the probability of choosing a red counter from a bag with \mbox{$3$} red and \mbox{$2$} blue counters in F/T form.
\task Write the probability of choosing a blue counter from a bag with \mbox{$4$} green and \mbox{$1$} blue counter in F/T form.
\task Write the probability of landing on \mbox{$A$} from a fair spinner with equal sectors labelled \mbox{$A$}, \mbox{$B$}, \mbox{$C$}, and \mbox{$D$}.
\task Write the probability of landing on a vowel from a fair spinner labelled \mbox{$A$}, \mbox{$B$}, \mbox{$C$}, \mbox{$D$}, \mbox{$E$}.
\task There are \mbox{$10$} marbles and \mbox{$7$} are yellow. Write the probability of picking yellow in F/T form.
\task There are \mbox{$8$} cards and \mbox{$2$} are stars. Write the probability of picking a star in F/T form.
\task Fill in the blank: for a fair die, \mbox{$P(\text{odd})=\frac{\square}{6}$}.
\task Fill in the blank: for one coin toss, \mbox{$P(\text{heads})=\frac{1}{\square}$}.
\task Which is greater: \mbox{$P(\text{rolling }1)$} or \mbox{$P(\text{rolling an even number})$} on a fair die?
\task Which is smaller: \mbox{$P(\text{heads})$} on a fair coin or \mbox{$P(\text{rolling }5)$} on a fair die?
\task A student says the probability of rolling a number less than \mbox{$7$} on a fair die is \mbox{$\frac{7}{6}$}. Are they correct?
\end{tasks}
