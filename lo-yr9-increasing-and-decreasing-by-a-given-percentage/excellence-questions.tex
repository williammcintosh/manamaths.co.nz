\begin{tasks}(3)
\task A student says increasing \mbox{$80$} by \mbox{$25\%$} gives \mbox{$85$}. Are they correct? Explain.
\task A student says decreasing \mbox{$120$} by \mbox{$15\%$} gives \mbox{$102$}. Are they correct? Explain.
\task Increase \mbox{$250$} by \mbox{$18\%$}.
\task Increase \mbox{$64$} by \mbox{$12.5\%$}.
\task Increase \mbox{$180$} by \mbox{$37.5\%$}.
\task Decrease \mbox{$240$} by \mbox{$17.5\%$}.
\task Decrease \mbox{$160$} by \mbox{$12.5\%$}.
\task Decrease \mbox{$320$} by \mbox{$37.5\%$}.
\task A jacket costs 120 dollars and is marked up by \mbox{$15\%$}. What is the new price?
\task A tablet costs 400 dollars and is discounted by \mbox{$22\%$}. What is the sale price?
\task A car value of 18,000 dollars falls by \mbox{$12\%$}. What is the new value?
\task A café increases a 6.40 dollar smoothie by \mbox{$25\%$}. What is the new price?
\task Fill in the blank: increasing \mbox{$96$} by \mbox{$12.5\%$} gives \mbox{$\square$}.
\task Fill in the blank: decreasing \mbox{$500$} by \mbox{$18\%$} gives \mbox{$\square$}.
\task Which is greater: increasing \mbox{$160$} by \mbox{$12.5\%$} or decreasing \mbox{$240$} by \mbox{$25\%$}? Show enough working to justify.
\task Which is smaller: increasing \mbox{$80$} by \mbox{$35\%$} or decreasing \mbox{$130$} by \mbox{$10\%$}? Explain.
\task A price is increased by \mbox{$10\%$} and then decreased by \mbox{$10\%$}. Does it return to the start? Use \mbox{$100$} as an example.
\task Explain why decreasing by \mbox{$25\%$} is not the same as multiplying by \mbox{$0.25$}.
\end{tasks}
