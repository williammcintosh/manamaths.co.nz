{\LARGE \textbf{Excellence}}
\begin{tasks}(3)
\task Write an expression for \mbox{$2$} more than three times \mbox{$x$}.
\task Write an expression for \mbox{$5$} less than double \mbox{$y$}.
\task Write an expression for half of \mbox{$n$} plus \mbox{$7$}.
\task Write an expression for \mbox{$4$} multiplied by the sum of \mbox{$p$} and \mbox{$3$}.
\task Write an expression for one third of \mbox{$m$} minus \mbox{$2$}.
\task Which is correct for "\mbox{$6$} less than \mbox{$a$}": \mbox{$a-6$} or \mbox{$6-a$}? Explain.
\task Which is correct for "the sum of \mbox{$x$} and \mbox{$5$}, then doubled": \mbox{$2(x+5)$} or \mbox{$2x+5$}? Explain.
\task Which does not belong: \mbox{$3x$}, \mbox{$x+x+x$}, \mbox{$x^3$}, \mbox{$2x+x$}?
\task Complete: a possible expression for "\mbox{$8$} more than half of \mbox{$t$}" is \rule{2.4cm}{0.15mm}.
\task Complete: a possible expression for "triple \mbox{$k$}, then subtract \mbox{$1$}" is \rule{2.4cm}{0.15mm}.
\task A student says \mbox{$2a+3$} means "double the sum of \mbox{$a$} and \mbox{$3$}". Are they correct? Explain.
\task A student says \mbox{$4(n-1)$} and \mbox{$4n-1$} mean the same thing. Are they correct? Explain.
\task Write an expression for the perimeter of a rectangle with width \mbox{$w$} and length \mbox{$w+2$}.
\task Write an expression for the total cost of \mbox{$x$} pens at \mbox{$3$} dollars each and one book costing \mbox{$5$} dollars.
\task Write an expression for the number of wheels on \mbox{$b$} bikes and \mbox{$s$} scooters if each has \mbox{$2$} wheels.
\task Explain why an expression can have a value even when it is not an equation.
\end{tasks}
