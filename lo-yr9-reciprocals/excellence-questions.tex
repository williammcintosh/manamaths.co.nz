\begin{tasks}(3)
\task A student says the reciprocal of \mbox{$\frac{4}{7}$} is \mbox{$\frac{7}{4}$}. Are they correct? Explain.
\task A student says the reciprocal of \mbox{$6$} is \mbox{$6$}. Are they correct? Explain.
\task Fill in both blanks: \mbox{$\frac{\square}{7}$} and \mbox{$\frac{7}{\square}$} are reciprocals.
\task Fill in both blanks: if \mbox{$\frac{5}{8}$} and \mbox{$\frac{a}{b}$} are reciprocals, then \mbox{$a=\rule{1.4cm}{0.15mm}$} and \mbox{$b=\rule{1.4cm}{0.15mm}$}.
\task Which pair are reciprocals: \mbox{$\frac{3}{8}$} and \mbox{$\frac{8}{3}$}, or \mbox{$\frac{3}{8}$} and \mbox{$\frac{3}{8}$}?
\task Write a number whose reciprocal is \mbox{$\frac{4}{9}$}.
\task Write a fraction whose reciprocal is \mbox{$3$}.
\task Complete: the reciprocal of \mbox{$\frac{m}{9}$} is \mbox{$\frac{9}{m}$}. If the reciprocal is \mbox{$\frac{9}{4}$}, then \mbox{$m=\rule{1.4cm}{0.15mm}$}.
\task If two numbers are reciprocals, what is their product?
\task Are \mbox{$\frac{6}{11}$} and \mbox{$\frac{11}{6}$} reciprocals? How do you know?
\task Which does not belong: \mbox{$\frac{2}{3}$}, \mbox{$\frac{3}{2}$}, \mbox{$\frac{4}{5}$}, \mbox{$\frac{5}{4}$}?
\task The reciprocal of a number is \mbox{$\frac{7}{12}$}. What is the number?
\task The reciprocal of a number is \mbox{$5$}. What is the number?
\task A student writes \mbox{$\frac{2}{9}\to\frac{9}{2}$} and \mbox{$9\to9$}. Explain why only one of these is correct.
\task Find a number whose reciprocal is smaller than \mbox{$\frac{1}{4}$}.
\task Explain why a number and its reciprocal switch places above and below the fraction bar.
\end{tasks}
